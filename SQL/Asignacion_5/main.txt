a)
SELECT COUNT(*) AS CantidadAutores
FROM AutorLibro
WHERE IdLibro = 1;

b)
SELECT COUNT(*) AS CantidadAutores
FROM AutorLibro al
JOIN Autor a ON al.IdAutor = a.IdAutor
JOIN AutorEditora ae ON a.IdAutor = ae.IdAutor
WHERE al.IdLibro = 1
AND ae.IdEditora = 1;

c) 
SELECT l.Titulo
FROM AutorLibro al
JOIN Libro l ON al.IdLibro = l.IdLibro
JOIN Editora e ON l.IdEditora = e.IdEditora
WHERE al.IdAutor = 1
AND e.IdEditora = 1;
d)
formato completo 
SELECT Nombre + ' ' + Apellido AS NombreAutorCompleto
FROM Autor
WHERE IdAutor = 1;

formato reducido
SELECT CONCAT(Nombre, ' ', LEFT(Apellido, 1), '.') AS NombreAutorReducido1
FROM Autor
WHERE IdAutor = 1;

formato reducido 2
SELECT CONCAT(Nombre, ' ', LEFT(Apellido, 1), '.', ' ', LEFT(Apellido, 2), '.') AS NombreAutorReducido2
FROM Autor
WHERE IdAutor = 1;

e)
SELECT STRING_AGG(CONCAT(Nombre, ' ', LEFT(Apellido, 1), '.', ' ', LEFT(Apellido, 2), '.'), ', ') AS AutoresLibro
FROM AutorLibro al
JOIN Autor a ON al.IdAutor = a.IdAutor
WHERE al.IdLibro = 1;

2)
a)
CREATE VIEW AutoresPorCasaEditora AS
SELECT l.IdLibro, l.Titulo, e.Nombre AS CasaEditora, a.Nombre AS NombreAutor, a.Apellido AS ApellidoAutor
FROM Libro l
JOIN Editora e ON l.IdEditora = e.IdEditora
JOIN AutorLibro al ON l.IdLibro = al.IdLibro
JOIN Autor a ON al.IdAutor = a.IdAutor;
b)
CREATE VIEW LibrosConCasasEditoraAutores AS
SELECT l.Titulo AS TituloLibro, e.Nombre AS CasaEditora, CONCAT(a.Nombre, ' ', a.Apellido) AS NombreAutor
FROM Libro l
JOIN Editora e ON l.IdEditora = e.IdEditora
JOIN AutorLibro al ON l.IdLibro = al.IdLibro
JOIN Autor a ON al.IdAutor = a.IdAutor;
